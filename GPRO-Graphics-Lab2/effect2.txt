//Effect #2: Makes circle that changes size and color
vec4 myEffect2(in vec2 fragCoord, in vec2 resolution)
{
    float radius = sin(iTime / 2.0) * 100.0 + 100.0; //Determines radius of the circle via time elapsed (uses sin to make repeatedly increase and decrease)
    vec2 circleCoord = vec2(resolution.x / 2.0, resolution.y / 2.0); //Circle is placed in the center of the screen
    vec4 circleColor = vec4(radius / 200.0, 1.0 - (radius / 200.0), 0.0, 1.0); //Changes color based on current size
    vec4 backgroundColor = vec4(1.0 - (radius / 200.0) , radius / 200.0, 1.0, 1.0); //Background color is opposite of circle (complementary)
    
    //Determines if the pixel is inside the circle and returns color based on it
    if(pow(fragCoord.x - circleCoord.x, 2.0) + pow(fragCoord.y - circleCoord.y, 2.0) > pow(radius, 2.0))
    {
        return vec4(backgroundColor);
    }
    
    return vec4(circleColor);
}
